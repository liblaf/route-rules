    def init(level: str | int = logging.NOTSET) -> None:

    logger.add(

        level=level,
        filter={"httpcore": logging.INFO, "httpx": logging.INFO},
    )
    logging.basicConfig(handlers=[InterceptHandler()], level=0, force=True)
